#include <cstdio>

double s;

int main() {
  scanf("%lf", &s);
  if (s >= 0 && s <= 2000) puts("Isento");
  else if (s >= 2000.01 && s <= 3000) printf("R$ %.2lf\n", (s - 2000) * 0.08);
  else if (s >= 3000.01 && s <= 4500) printf("R$ %.2lf\n", (s - 3000) * 0.18 + 1000 * 0.08);
  else if (s >= 4500) printf("R$ %.2lf\n", (s - 4500) * 0.28 + 1500 * 0.18 + 1000 * 0.08);
  return 0;
}
